Summary and Remarks¶

By checking the source code from sikit-learn and libsvm, we find that actually, the decision_function_shape='ovr' don't take effect in the SVC() interface. Actually, no matter how u set, the SVC() only works as ovo since the source code is libsvm-svm.cpp. https://github.com/scikit-learn/scikit-learn/blob/main/sklearn/svm/src/libsvm/svm.cpp

Hence, the only way we really using OVR in sklearn interface is that, first, we make a one-hot encoding(Directly input y like[0, 1, 2, ...] also ok) for reshape the target/Y data. Then, using the OneVsRestClassifier() interface, you can regard that the OVR(SVC(#the decision_function_shape still take no effect.#)) actually trained 3 SVC-ovo for binary classification (Please also note that for binary classification, ovo is definitely same to ovr).

From Prof Wu and Prof Fan: Both OvO/OvR are acceptable to finish this assignment. Students using any methods are ok(Even they actually do not notice the hyper parameters "decision_function_shape", but the basic idea for multiclass classification(OvO/OvR) is very important). Here the scikit-learn API has these unfixed bugs. Hope this .ipynb and some tests from TA can help address their confusion.

def plot_hyperplane(clf, min_x, max_x, linestyle, label, decision_function_shape='ovr', class_ = None):
    
    if decision_function_shape=='ovr':
        # get the separating hyperplane
        w = clf.coef_[0]
        a = -w[0] / w[1]
        xx = np.linspace(min_x - 5, max_x + 5)  # make sure the line is long enough
        yy = a * xx - (clf.intercept_[0]) / w[1]
        plt.plot(xx, yy, linestyle, label=label)
        
    if decision_function_shape=='ovo':
        w = clf.coef_[class_]
        a = -w[0] / w[1]
        xx = np.linspace(min_x - 5, max_x + 5)  # make sure the line is long enough
        yy = a * xx - (clf.intercept_[class_]) / w[1]
        plt.plot(xx, yy, linestyle, label=label)


def plot_subfigure(X, Y, subplot, title, decision_function_shape='ovr'):
        
    transform = "pca"
    X = PCA(n_components=2,random_state = 0).fit_transform(X)

    min_x = np.min(X[:, 0])
    max_x = np.max(X[:, 0])
    min_y = np.min(X[:, 1])
    max_y = np.max(X[:, 1])

    if decision_function_shape=='ovr':
        classif = OneVsRestClassifier(SVC(kernel="linear", random_state=0,decision_function_shape='ovr'))
        classif.fit(X, Y)
    if decision_function_shape=='ovo':
        classif = SVC(kernel="linear", random_state=0,decision_function_shape='ovo')
        classif.fit(X, Y)

    plt.subplot(1, 2, subplot)
    plt.title(title)
    

    zero_class = np.where(Y==0)
    one_class = np.where(Y==1)
    two_class = np.where(Y==2)
        
    plt.scatter(X[:, 0], X[:, 1], s=40, c="gray", edgecolors=(0, 0, 0))
    plt.scatter(
        X[zero_class, 0],
        X[zero_class, 1],
        s=160,
        edgecolors="b",
        facecolors="none",
        linewidths=2,
        label="Class 1",
    )
    plt.scatter(
        X[one_class, 0],
        X[one_class, 1],
        s=80,
        edgecolors="orange",
        facecolors="none",
        linewidths=2,
        label="Class 2",
    )
    plt.scatter(
        X[two_class, 0],
        X[two_class, 1],
        s=80,
        edgecolors="red",
        facecolors="none",
        linewidths=2,
        label="Class 3",
    )

    if decision_function_shape=='ovr':
        plot_hyperplane(
            classif.estimators_[0], min_x, max_x, "k--", "Boundary\nfor class 1"
        )
        plot_hyperplane(
            classif.estimators_[1], min_x, max_x, "k-.", "Boundary\nfor class 2"
        )
        plot_hyperplane(
            classif.estimators_[2], min_x, max_x, "k-*", "Boundary\nfor class 3"
        )
    if decision_function_shape=='ovo':
        plot_hyperplane(
            classif, min_x, max_x, "k--", "Boundary\nfor class 1",decision_function_shape='ovo',class_ = 0
        )
        plot_hyperplane(
            classif, min_x, max_x, "k-.", "Boundary\nfor class 2",decision_function_shape='ovo',class_ = 1
        )
        plot_hyperplane(
            classif, min_x, max_x, "k-*", "Boundary\nfor class 3",decision_function_shape='ovo',class_ = 2
        )
    plt.xticks(())
    plt.yticks(())

    plt.xlim(min_x - 0.5 * max_x, max_x + 0.5 * max_x)
    plt.ylim(min_y - 0.5 * max_y, max_y + 0.5 * max_y)
    
    plt.xlabel("First principal component")
    plt.ylabel("Second principal component")
    plt.legend(loc="upper left")
